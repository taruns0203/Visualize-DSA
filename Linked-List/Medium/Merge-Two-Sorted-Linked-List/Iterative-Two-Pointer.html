üíª Core Algorithm
while (list1 !== null && list2 !== null) {
    if (list1.val <= list2.val) {
        tail.next = list1;    // Link smaller node
        list1 = list1.next;   // Advance pointer
    } else {
        tail.next = list2;
        list2 = list2.next;
    }
    tail = tail.next;       // Move tail forward
}
tail.next = list1 ?? list2;  // Attach remainder
return dummy.next;
